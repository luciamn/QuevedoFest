# AREA DE PRODUCCION
    
    CREATE DATABASE AreaProduccion;

## Tabla Catering
    CREATE TABLE Catering (
    Telefono VARCHAR(9),
    Nombre VARCHAR(40),
    ID VARCHAR(5) NOT NULL,
    Direccion VARCHAR(40)
    )

## Tabla Menu 
    CREATE TABLE  Menu (
    Nombre VARCHAR(40),
    Tipo Nombre VARCHAR(40),
    ID VARCHAR(5) NOT NULL
    )
